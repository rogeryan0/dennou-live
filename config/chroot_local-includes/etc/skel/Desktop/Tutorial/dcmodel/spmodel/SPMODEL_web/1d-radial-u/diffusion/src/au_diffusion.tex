この文章では, チェビシェフ関数展開を利用した
球および円盤領域の動径方向 1 次元計算のための SPMODEL ライブラリ(spml)の
モジュール au\_module のテストのために, 
球および円筒領域での動径 1 次元拡散型方程式の
さまざまな境界条件に対する解を(できるだけ)解析的に
求めることを行う. 

  \section{球領域の動径 1 次元拡散問題}

    半径 a の球領域での拡散問題を考える. 支配方程式は
    %
    \begin{equation}
      \Deqlab{球拡散方程式}
      \DP{\phi}{t} = \kappa \frac{1}{r^2}\DP{}{r}\left(r^2\DP{\phi}{r}\right).
    \end{equation}
    %
    $r (0\le r \le a)$ は動径座標, $\kappa$ は拡散率(正数)である. 
    $\phi(r,t)$ は拡散する物理量であり, 例えば温度, 組成濃度, 磁場などである.

    \subsection{一般解}

      時間に関して指数型の解を仮定し, 
      %
      \begin{equation}
        \Deqlab{球面調和函数展開}
        \phi(r,t)=\tilde{\phi}(r)e^{\sigma t}
      \end{equation}
      %
      これを支配方程式に代入すると, 
      %
      \begin{equation}
        \Deqlab{拡散方程式球面調和函数展開}
          \sigma \tilde{\phi} = 
              \kappa \frac{1}{r^2}\DD{}{r}\left(r^2\DD{\tilde{\phi}}{r}\right).
      \end{equation}
      %
      これをを変形すると
      %
      \begin{displaymath}
        \frac{1}{r^2}\DD{}{r}\left(r^2\DD{\tilde{\phi}}{r}\right)
           + \frac{-\sigma}{\kappa}\tilde{\phi},
      \end{displaymath}
      %
      さらに動径座標を
      \begin{equation}
        \tilde{r} = \sqrt{\frac{-\sigma}{\kappa}}r, \quad
        r  = \sqrt{\frac{\kappa}{-\sigma}}\tilde{r},
      \end{equation}
      %
      と変形すると,
      %
      \begin{displaymath}
        \frac{1}{\tilde{r}^2}\DD{}{\tilde{r}}
             \left(\tilde{r}^2\DD{\tilde{\phi}}{\tilde{r}}\right)
          +\tilde{\phi}=0.
      \end{displaymath}
      %
      この微分方程式は 0 次の球ベッセルの微分方程式であり, その一般解は
      $0$ 次の球ベッセル函数 ${\cal J}_0(\tilde{r}),{\cal N}_0(\tilde{r})$ の
      線形結合として表される.


    \subsection{ディリクレ境界条件の場合}

      境界条件は, 原点 $r=0$ で特異性がなく, 
      外側境界 $r=a$ でディリクレ条件 $\tilde{\phi}=0$ を考えることにする
      %
      \footnote{
        一般的には, ディリクレ条件だと $\tilde{\phi}=A$, 
        しかしながら, 
        ディリクレ条件の場合は定数部分を $\tilde{\phi}=A+\tilde{\phi}' $
        と分離できるので結局値が 0 に対する解を解けば良いことになる. 
        ノイマン条件の場合は...
      }.  
      原点の境界条件から
      %
      \begin{equation}
        \tilde{\phi} = A {\cal J}_0(\tilde{r}) 
                     = A\frac{\sin\tilde{r}}{\tilde{r}}
      \end{equation}
      %
      $r=a$ の境界条件から
      %
      \begin{equation}
        \tilde{a} = \sqrt{\frac{-\sigma}{\kappa}}a = n\pi, \quad n=1,2,\ldots
      \end{equation}
      %
      すなわち,
      %
      \begin{equation}
        \sigma = -\frac{\kappa}{a^2}n^2\pi^2, \quad n=1,2,\ldots
      \end{equation}
      %
      よって解は
      \begin{equation}
        \phi = \sum_{n=1}^\infty A_n {\cal J}_0(\frac{n\pi}{a}r)
               \exp(-\frac{\kappa}{a^2}n^2\pi^2 t).
           = \sum_{n=1}^\infty A_n \frac{\sin(\frac{n\pi}{a}r)}{\frac{n\pi}{a}r}
               \exp(-\frac{\kappa}{a^2}n^2\pi^2 t).
      \end{equation}
      %
      係数 $A_n$ は初期条件からさだまる. 

    \subsection{ノイマン境界条件の場合}

      境界条件は, 原点 $r=0$ で特異性がなく, 
      外側境界 $r=a$ でノイマン条件 $\Ddsty\DD{\tilde{\phi}}{r}=0$ を
      考えることにする
      %
      \footnote{
        一般的なノイマン条件だと $\Ddsty\DD{\tilde{\phi}}{r}=A$. 
      }.  
      原点の境界条件から
      %
      \begin{equation}
        \tilde{\phi} = A {\cal J}_0(\tilde{r}) 
                     = A\frac{\sin\tilde{r}}{\tilde{r}}, 
      \end{equation}
      %
      $r=a$ の境界条件から
      %
      \begin{displaymath}
        \DD{\tilde{\phi}}{r} = A {\cal J}'_0(\tilde{r}) 
          = A\frac{\tilde{r}\cos\tilde{r}-\sin\tilde{r}}{\tilde{r}^2}=0,
      \end{displaymath}
      %
      したがって
      \begin{equation}
          \tilde{a}\cos\tilde{a}-\sin\tilde{a}=0. 
      \end{equation}
      %
      $x\cos x - \sin x =0 $ の解を $\alpha_n$ と求まったとすると, 
      %
      \begin{displaymath}
        \sqrt{\frac{-\sigma}{\kappa}}a=\alpha_n, \quad
        \sqrt{\frac{-\sigma}{\kappa}}=\frac{\alpha_n}{a}, \quad
        \sigma= -\kappa\frac{\alpha_n^2}{a^2}, \quad
      \end{displaymath}
      %
      \begin{equation}
        \phi = A_n\frac{\sin\frac{\alpha_n r}{a}}{\frac{\alpha_n r}{a}}
              \exp\left(-\kappa\frac{\alpha_n^2}{a^2}t\right)
      \end{equation}
